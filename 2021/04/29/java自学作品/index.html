    if(!(str != null)){
        throw new RuntimeException("字符串不能为空");
    }else if(!(start >= 0)){
        throw new RuntimeException("起始值要大于等于零");
    }else if(!(start < end)){
        throw new RuntimeException("起始值要小于结束值");
    }else if(!(end < str.length())){

    }

    char[] chars = str.toCharArray();
    char temp = ' ';
    //当start大于或者等于end时，停止交换
    for (int i = start, j = end; i < j; i++,j--) {
        temp = chars[i];
        chars[i] = chars[j];
        chars[j] = temp;
    }
    String s = new String(chars);
    return s;
